🌳 What Is a Binary Tree?

A binary tree is a tree where each node has at most two children: left and right. The top node is called the root.

class TreeNode:
    def __init__(self, val=0, left=None, right=None):
        self.val = val
        self.left = left
        self.right = right

Each node stores a value and pointers to left/right children.

        1
       /  \
      2    3
     / \   \
    4   5    6

 # TreeNode(1,
 #   TreeNode(2, TreeNode(4), TreeNode(5)),
 #   TreeNode(3, None, TreeNode(6))
 # )
🔑 Key terms: Root (top), Leaf (no children), Depth (edges from root), Height (edges to deepest leaf)

📐 Inorder Traversal (Left → Root → Right)

Visit the left subtree first, then the node itself, then the right subtree. On a BST, this gives sorted order.

def inorder(root):
    if root is None:
        return
    # 1. Go left
    inorder(root.left)
    # 2. Visit node
    print(root.val, end=" ")
    # 3. Go right
    inorder(root.right)
 # For tree:
 #   1 -> L:2 -> L:4
 #        -> R:5
 #       R:3 -> R:6

 inorder(root)
 # Output: 4 2 5 1 3 6
 # Path: 4→2→5→1→3→6

 # LeetCode #94
 def inorderTraversal(root):
    res = []
    def dfs(node):
        if not node:
            return
        dfs(node.left)
        res.append(node.val)
        dfs(node.right)
    dfs(root)
    return res

📐 Preorder Traversal (Root → Left → Right)

Visit the current node first, then the left subtree, then the right subtree. Useful for copying or serializing a tree.

def preorder(root):
    if root is None:
        return
    # 1. Visit node first
    print(root.val, end=" ")
    # 2. Then go left
    preorder(root.left)
    # 3. Then go right
    preorder(root.right)
 # Same tree:
 #   1 -> L:2 -> L:4
 #        -> R:5
 #       R:3 -> R:6

 preorder(root)
 # Output: 1 2 4 5 3 6
 # Path: 1→2→4→5→3→6

 # Use case: tree serialization
 def serialize(root):
    if not root:
        return "#"
    left = serialize(root.left)
    right = serialize(root.right)
    return f"{root.val},{left},{right}"

📐 Postorder Traversal (Left → Right → Root)

Visit left, then right, then root. Used for deleting trees or computing values that depend on children first (like tree height).

def postorder(root):
    if root is None:
        return
    # 1. Go left
    postorder(root.left)
    # 2. Go right
    postorder(root.right)
    # 3. Visit node last
    print(root.val, end=" ")
 # Same tree:
 #   1 -> L:2 -> L:4
 #        -> R:5
 #       R:3 -> R:6

 postorder(root)
 # Output: 4 5 2 6 3 1
 # Path: 4→5→2→6→3→1

 # Use case: tree deletion
 # (delete children before parent)
 def delete_tree(root):
    if not root:
        return
    delete_tree(root.left)
    delete_tree(root.right)
    root.left = root.right = None
    # now safe to free root

Same tree, three different output orders. Memorize by when you visit the root.

Preorder

Root → Left → Right

Postorder

4 5 2 1 3

Left → Right → Root

Tree deletion

💡 Mnemonic: The name tells you when the root is visited. "Pre" = before children, "In" = between children, "Post" = after children.

📏 Maximum Depth of Binary Tree

The max depth (or height) is the number of nodes along the longest path from root to a leaf. LeetCode #104 (Easy).

# Recursive: postorder logic
# (compute children first, then root)
def maxDepth(root) -> int:
    if not root:
        return 0
    left = maxDepth(root.left)
    right = maxDepth(root.right)
    return 1 + max(left, right)

# Time: O(n) — visit every node once
# Space: O(h) — call stack = tree height
 # Walkthrough for:
 #     3
 #    / \
 #   9  20
 #      / \
 #    15   7

 maxDepth(3)
   left = maxDepth(9) = 1
   right = maxDepth(20)
     left = maxDepth(15) = 1
     right = maxDepth(7) = 1
     right = 1 + max(1,1) = 2
   return 1 + max(1,2) = 3

def maxDepth(root):
    if not root:
        return 0
    return 1 + max(
        maxDepth(root.left),
        maxDepth(root.right)

Ex 3: Build Tree (BFS)

def build_tree(values):
    if not values or values[0] is None:
        return None
    root = TreeNode(values[0])
    q = collections.deque([root])
    i = 1
    while q and i < len(values):
        node = q.popleft()
        if i < len(values) and values[i] is not None:
            node.left = TreeNode(values[i])
            q.append(node.left)
        i += 1
        if i < len(values) and values[i] is not None:
            node.right = TreeNode(values[i])
            q.append(node.right)
        i += 1
    return root

🎯 Key Takeaways

1. Node Structure

Binary tree node = val + left + right. None = empty subtree.

2. Three Traversals

Inorder: L→Root→R. Preorder: Root→L→R. Postorder: L→R→Root.

3. Recursion Pattern

Base case: if not root: return. Recurse on children. Process node at the right time.

4. Next Steps

Level-order (BFS), iterative traversal, tree construction from traversals.
